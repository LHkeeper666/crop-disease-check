## 2. 功能点关系

| 功能 | 角色 | 说明 |
|------|------|------|
| 视频流抽帧推理 | 基础能力 | 从RTSP流抓一帧图片，调推理服务，返回bbox坐标 |
| 定时更新 | 触发机制 | 决定"多久调一次抽帧推理"（每10秒/30秒/1分钟） |
| 带检测框的视频 | 输出产物 | HLS视频流 + Canvas叠加层实时绘制检测框 |
| 多窗口 | 展示层 | 同时对多个摄像头执行抽帧推理，结果渲染到各自窗口 |

### 时序流程

```
时间线 ──────────────────────────────────────────────►

              │           │           │
              ▼           ▼           ▼

              │           │           │

              ▼           ▼           ▼
```

## 3. 后端接口设计

### 3.1 启动/停止实时监测

**POST** `/api/v1/camera/{id}/monitor`

```json
// 请求
{
  "enabled": true,
  "intervalSeconds": 5,
  "confidence": 0.5,
  "useSubStream": false
}

// 响应
{
  "code": 200,
  "message": "实时监测已启动"
}
```

| 字段 | 类型 | 默认值 | 说明 |
|------|------|--------|------|
| enabled | boolean | true | 启用/禁用监测 |
| intervalSeconds | int | 5 | 抽帧间隔(秒)，最小3秒 |
| confidence | float | 0.5 | 置信度阈值 |
| useSubStream | boolean | false | 是否使用子码流 |

### 5.3 HLS视频播放

```typescript
import Hls from 'hls.js';

function initHlsPlayer(videoEl: HTMLVideoElement, cameraId: string) {
  if (!Hls.isSupported()) {
    // 降级到原生HLS支持（Safari）
    videoEl.src = `/api/v1/stream/${cameraId}.m3u8`;
    return;
  }

  const hls = new Hls({
    liveSyncDurationCount: 1,    // 减少直播延迟
    liveMaxLatencyDurationCount: 3,
    maxBufferLength: 5,
  });

  hls.loadSource(`/api/v1/stream/${cameraId}.m3u8`);
  hls.attachMedia(videoEl);

  hls.on(Hls.Events.MANIFEST_PARSED, () => {
    videoEl.play();
  });

  hls.on(Hls.Events.ERROR, (event, data) => {
    if (data.fatal) {
      console.error('HLS致命错误:', data.type, data.details);
      // 重试逻辑
      if (data.type === Hls.ErrorTypes.NETWORK_ERROR) {
        hls.startLoad();
      } else if (data.type === Hls.ErrorTypes.MEDIA_ERROR) {
        hls.recoverMediaError();
      }
    }
  });

  return hls;
}
```

### 5.4 WebSocket订阅

```typescript
import SockJS from 'sockjs-client';
import { Client } from '@stomp/stompjs';

function initWebSocket(cameraId: string, onDetection: Function) {
  const client = new Client({
    webSocketFactory: () => new SockJS('/ws'),
    reconnectDelay: 5000,
    heartbeatIncoming: 4000,
    heartbeatOutgoing: 4000,
  });

  client.onConnect = () => {
    // 订阅特定摄像头的检测结果
    client.subscribe(`/topic/camera/${cameraId}/detect`, (message) => {
      const data = JSON.parse(message.body);
      onDetection(data);
    });
  };

  client.activate();
  return client;
}
```

### 5.5 Canvas绘制检测框

```typescript
interface Detection {
  type: 'disease' | 'pest';
  classId: number;
  className: string;
  nameCn: string;
  confidence: number;
  bbox: {
    x: number;
    y: number;
    width: number;
    height: number;
  };
}

function drawDetections(
  canvas: HTMLCanvasElement,
  detections: Detection[],
  frameWidth: number,   // 抽帧分辨率宽度（640）
  frameHeight: number   // 抽帧分辨率高度（640）
) {
  const ctx = canvas.getContext('2d')!;
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  // 坐标映射：抽帧分辨率 → 视频显示尺寸
  const scaleX = canvas.width / frameWidth;
  const scaleY = canvas.height / frameHeight;

  detections.forEach(det => {
    const x = det.bbox.x * scaleX;
    const y = det.bbox.y * scaleY;
    const w = det.bbox.width * scaleX;
    const h = det.bbox.height * scaleY;

    // 样式：病害红色，虫害蓝色
    const color = det.type === 'disease' ? '#ff4444' : '#4488ff';

    // 画检测框
    ctx.strokeStyle = color;
    ctx.lineWidth = 2;
    ctx.strokeRect(x, y, w, h);

    // 画标签背景
    const label = `${det.nameCn} ${(det.confidence * 100).toFixed(0)}%`;
    ctx.font = '12px sans-serif';
    const textWidth = ctx.measureText(label).width;
    ctx.fillStyle = color;
    ctx.fillRect(x, y - 20, textWidth + 10, 20);

    // 画标签文字
    ctx.fillStyle = '#ffffff';
    ctx.fillText(label, x + 5, y - 5);
  });
}
```

### 5.6 完整组件示例

```vue
<template>
  <div class="camera-window" :style="{ width: width + 'px', height: height + 'px' }">
    <video ref="videoRef" autoplay muted></video>
    <canvas ref="canvasRef" :width="width" :height="height"></canvas>
    <div class="status-bar">
      <span class="name">{{ cameraName }}</span>
      <span :class="['status', status.toLowerCase()]">{{ status }}</span>
      <span class="count">病害: {{ diseaseCount }} | 虫害: {{ pestCount }}</span>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted, onUnmounted } from 'vue';
import Hls from 'hls.js';
import SockJS from 'sockjs-client';
import { Client } from '@stomp/stompjs';

const props = defineProps<{
  cameraId: string;
  cameraName: string;
  width?: number;
  height?: number;
}>();

const videoRef = ref<HTMLVideoElement>();
const canvasRef = ref<HTMLCanvasElement>();
const status = ref('OFFLINE');
const diseaseCount = ref(0);
const pestCount = ref(0);

let hls: Hls | null = null;
let stompClient: Client | null = null;

function initHls() {
  if (!videoRef.value) return;

  if (Hls.isSupported()) {
    hls = new Hls({
      liveSyncDurationCount: 1,
      liveMaxLatencyDurationCount: 3,
      maxBufferLength: 5,
    });
    hls.loadSource(`/api/v1/stream/${props.cameraId}.m3u8`);
    hls.attachMedia(videoRef.value);
    hls.on(Hls.Events.MANIFEST_PARSED, () => {
      videoRef.value?.play();
      status.value = 'ONLINE';
    });
    hls.on(Hls.Events.ERROR, (_event, data) => {
      if (data.fatal) {
        status.value = 'FAULT';
      }
    });
  } else if (videoRef.value.canPlayType('application/vnd.apple.mpegurl')) {
    videoRef.value.src = `/api/v1/stream/${props.cameraId}.m3u8`;
    status.value = 'ONLINE';
  }
}

function initWebSocket() {
  stompClient = new Client({
    webSocketFactory: () => new SockJS('/ws'),
    reconnectDelay: 5000,
  });

  stompClient.onConnect = () => {
    stompClient?.subscribe(`/topic/camera/${props.cameraId}/detect`, (message) => {
      const data = JSON.parse(message.body);
      diseaseCount.value = data.diseaseCount || 0;
      pestCount.value = data.pestCount || 0;

      if (canvasRef.value) {
        drawDetections(
          canvasRef.value,
          data.detections,
          data.frameWidth,
          data.frameHeight
        );
      }
    });
  };

  stompClient.activate();
}

function drawDetections(
  canvas: HTMLCanvasElement,
  detections: any[],
  frameWidth: number,
  frameHeight: number
) {
  const ctx = canvas.getContext('2d')!;
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  const scaleX = canvas.width / frameWidth;
  const scaleY = canvas.height / frameHeight;

  detections.forEach(det => {
    const x = det.bbox.x * scaleX;
    const y = det.bbox.y * scaleY;
    const w = det.bbox.width * scaleX;
    const h = det.bbox.height * scaleY;

    const color = det.type === 'disease' ? '#ff4444' : '#4488ff';

    ctx.strokeStyle = color;
    ctx.lineWidth = 2;
    ctx.strokeRect(x, y, w, h);

    const label = `${det.nameCn} ${(det.confidence * 100).toFixed(0)}%`;
    ctx.font = '12px sans-serif';
    const textWidth = ctx.measureText(label).width;

    ctx.fillStyle = color;
    ctx.globalAlpha = 0.7;
    ctx.fillRect(x, y - 20, textWidth + 10, 20);
    ctx.globalAlpha = 1;

    ctx.fillStyle = '#ffffff';
    ctx.fillText(label, x + 5, y - 5);
  });
}

onMounted(() => {
  initHls();
  initWebSocket();
});

onUnmounted(() => {
  hls?.destroy();
  stompClient?.deactivate();
});
</script>
```

```
│                 │           │               │

│                 │           │               │
└─────────────────┘           └───────────────┘

scaleX = 480 / 640 = 0.75
scaleY = 360 / 640 = 0.5625

scaled_x = 120 * 0.75 = 90
scaled_y = 80 * 0.5625 = 45
scaled_w = 200 * 0.75 = 150
scaled_h = 150 * 0.5625 = 84
```

```java
// RTSP抽帧（JavaCV）
private byte[] captureFrameFromRtsp(String rtspUrl) throws Exception {
    FFmpegFrameGrabber grabber = new FFmpegFrameGrabber(rtspUrl);
    grabber.setOption("rtsp_transport", "tcp");
    grabber.setImageWidth(640);
    grabber.setImageHeight(640);
    grabber.start();

    Frame frame = grabber.grabImage();
    BufferedImage image = new Java2DFrameConverter().convert(frame);

    ByteArrayOutputStream baos = new ByteArrayOutputStream();
    ImageIO.write(image, "jpg", baos);
    return baos.toByteArray();
}
```
